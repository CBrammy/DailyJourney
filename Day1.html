        import java.io.BufferedReader;
        import java.io.FileReader;
        
        
        public class Main {
        
        public static void main(String[] args) throws Exception{
        // Each item on a line
        // each elves inventory divided by blank line
                
        
        BufferedReader br = new BufferedReader(new FileReader("src/sample.txt"));
        
        
        int calorie = 0;
        int mostCalories = 0;
        String line;
        
        while ((line = br.readLine()) != null){
             String cal = line.trim();
             if (cal.isBlank()) {
                 if (mostCalories < calorie) {
                     mostCalories = calorie;
                 }
                 calorie = 0;
                 continue;
             }
             calorie += Integer.parseInt(cal);
         }
        if (mostCalories < calorie) {
             mostCalories = calorie;
             }
             
        System.out.print(mostCalories);
        
        }
        }
            
        

    Takes in a txt file, for every line adds the sum to an integer.
    When the next line is blank, creates a sum and stores that in calorie.
    If mostCalories is less than calories, mostCalories takes the sum of calorie.
    Iterates through the entire sample.txt to find the biggest set of calories before a blank line.
